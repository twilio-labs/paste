export const meta = {
  title: '404',
  description: 'Error page',
  slug: '/404/',
};

import {Error404} from '../../components/empty-state/Error404';
import DefaultLayout from '../../layouts/DefaultLayout';
import {getNavigationData, getAllPackages} from '../../utils/api';

export default DefaultLayout;

export const getStaticProps = async () => {
  const navigationData = await getNavigationData();
  const packageData = await getAllPackages();
  return {
    props: {
      navigationData,
      packageData,
    },
  };
};

<Error404
  componentList={props.packageData.allPasteComponent}
  layoutList={props.packageData.allPasteLayout}
  primitiveList={props.packageData.allPastePrimitive}
/>
