## Installation

```bash
yarn add @twilio-paste/ai-chat-log - or - yarn add @twilio-paste/core
```

## Usage

```jsx
import {
  AIChatLog,
  AIChatMessage,
  AIChatMessageAuthor,
  AIChatMessageBody,
  AIChatMessageFeedback,
  AIChatMessageLoading,
  AIChatMessageMeta,
} from "@twilio-paste/core/ai-chat-log";

export const Basic = () => {
  return (
    <AIChatLog>
      <AIChatMessage variant="user">
        <AIChatMessageAuthor aria-label="You said">Gibby Radki</AIChatMessageAuthor>
        <AIChatMessageBody>
          Hi, I'm getting errors codes when sending an SMS.
        </AIChatMessageBody>
      </AIChatMessage>
      <AIChatMessage variant="bot">
        <AIChatMessageAuthor aria-label="AI said">Good Bot</AIChatMessageAuthor>
        <AIChatMessageBody>
          Error codes can be returned from various parts of the process. What error codes are you encountering?
        </AIChatMessageBody>
        <AIChatMessageActionGroup aria-label="Quick actions available:">
          <AIChatMessageAction>
            <Button variant="reset" size="reset" onClick={() => {}} >
              21608
            </Button>
          </AIChatMessageAction>
          <AIChatMessageAction>
            <Button variant="reset" size="reset" onClick={() => {}}>
              30007
            </Button>
          </AIChatMessageAction>
          <AIChatMessageAction>
            <Button variant="reset" size="reset" onClick={() => {}}>
              30009
            </Button>
          </AIChatMessageAction>
        </AIChatMessageActionGroup>
      </AIChatMessage>
    </AIChatLog>
  );
};
```
