## Usage Guide

### API

#### Installation

```bash
yarn add @twilio-paste/Status - or - yarn add @twilio-paste/core
```

#### Usage

```jsx
import {
  StatusMenu,
  StatusMenuBadge,
  StatusMenuItem,
  StatusMenuItemChild,
  useStatusMenuState,
} from '@twilio-paste/core/status';

const PreferencesMenu = () => {
  const menu = useStatusMenuState();
  return (
    <>
      <StatusMenuBadge {...menu} variant="ProcessErro">
        Preferences
      </StatusMenuBadge>
      <StatusMenu {...menu} aria-label="Preferences">
        <StatusMenuItem {...menu}>Settings</StatusMenuItem>
        <StatusMenuItem {...menu} disabled>
          Extensions
        </StatusMenuItem>
        <StatusMenuItem {...menu}>Keyboard shortcuts</StatusMenuItem>
      </StatusMenu>
    </>
  );
};
```
