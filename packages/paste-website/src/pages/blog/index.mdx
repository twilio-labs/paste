export const meta = {
  title: 'Articles',
  description: 'Articles published by the Design Systems team at Twilio',
  slug: '/blog/',
};

import {PageHeader, PageHeaderDetails, PageHeaderHeading} from '@twilio-paste/page-header';

import {ArticleList} from '../../components/ArticleList';
import DefaultLayout from '../../layouts/DefaultLayout';
import {getArticles, getNavigationData} from '../../utils/api';

export default DefaultLayout;

export const getStaticProps = async () => {
  const navigationData = await getNavigationData();
  const articles = await getArticles();
  return {
    props: {
      articles: articles,
      navigationData,
    },
  };
};

<PageHeader>
  <PageHeaderDetails>
    <PageHeaderHeading>Articles and Posts about Paste</PageHeaderHeading>
  </PageHeaderDetails>
</PageHeader>

<ArticleList articles={props.articles} />
